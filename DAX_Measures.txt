// =====================================================================
// DAX MEASURES — create these in a dedicated "_Measures" table
// (Modeling → New Table → = ROW("Measures Table", BLANK())  then hide the column)
// All measures reference the star schema table/column names exactly as
// defined in the Power Query steps.
// =====================================================================

Total Sales = SUM(Fact_OrderDetails[Sales])

Total Orders = DISTINCTCOUNT(Fact_OrderDetails[OrderID])

Total Order Items = COUNTROWS(Fact_OrderDetails)

Total Quantity Sold = SUM(Fact_OrderDetails[Quantity])

Average Order Value = DIVIDE([Total Sales], [Total Orders])

Total Benefit = SUM(Fact_OrderDetails[BenefitPerOrder])

Profit Margin % = DIVIDE([Total Benefit], [Total Sales])

Total Discount Given = SUM(Fact_OrderDetails[OrderItemDiscount])

Avg Discount Rate = AVERAGE(Fact_OrderDetails[DiscountRate])

Late Delivery Count =
CALCULATE(
    DISTINCTCOUNT(Fact_OrderDetails[OrderID]),
    Dim_Order[LateDeliveryRisk] = 1
)

Late Delivery % =
DIVIDE(
    [Late Delivery Count],
    [Total Orders]
)

Avg Shipping Days (Scheduled) = AVERAGE(Dim_Order[DaysForShipmentScheduled])

Avg Shipping Days (Real) = AVERAGE(Dim_Order[DaysForShippingReal])

Shipping Delay (Days) = [Avg Shipping Days (Real)] - [Avg Shipping Days (Scheduled)]

Total Customers = DISTINCTCOUNT(Fact_OrderDetails[CustomerID])

Sales Per Customer = DIVIDE([Total Sales], [Total Customers])

Sales LY =
CALCULATE([Total Sales], SAMEPERIODLASTYEAR(Dim_Date[Date]))

Sales Growth % =
DIVIDE([Total Sales] - [Sales LY], [Sales LY])

Shipping Date Sales =
CALCULATE(
    [Total Sales],
    USERELATIONSHIP(Dim_Date[DateKey], Fact_OrderDetails[ShippingDateKey])
)

// ---- Notes ----
// - "Total Sales" flows through the ACTIVE relationship: Dim_Date[DateKey] -> Fact_OrderDetails[OrderDateKey]
// - Use "Shipping Date Sales" (or wrap any measure with the same USERELATIONSHIP pattern)
//   whenever you need shipping-date-based analysis instead of order-date-based.
// - "Late Delivery %" and shipping-day measures pull from Dim_Order because those
//   attributes live at order grain, not order-item grain — RELATED works automatically
//   since Dim_Order[OrderID] -> Fact_OrderDetails[OrderID] is a valid 1-to-many path.
